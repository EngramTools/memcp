/// Validate every tool's parameters schema is well-formed JSON Schema.
/// Call at server startup (RESEARCH Pitfall 7).
pub fn validate_tool_schemas(tools: &[Tool]) -> Result<(), ReasoningError> {
    for t in tools {
        jsonschema::validator_for(&t.parameters)
            .map_err(|e| ReasoningError::BadToolSchema(t.name.clone(), e.to_string()))?;
    }
    Ok(())
}

/// Error-wrapping helper: build a ToolResult { is_error: true } from any displayable error.
fn err_result(call: &ToolCall, msg: impl std::fmt::Display) -> ToolResult {
    ToolResult {
        tool_call_id: call.id.clone(),
        content: msg.to_string(),
        is_error: true,
    }
}

fn ok_result(call: &ToolCall, content: String) -> ToolResult {
    ToolResult { tool_call_id: call.id.clone(), content, is_error: false }
}

/// Dispatch one tool call against the caller context. Never panics;
/// always returns a ToolResult (error wrapped via is_error).
pub async fn dispatch_tool(call: &ToolCall, ctx: &AgentCallerContext) -> ToolResult {
    match call.name.as_str() {
        "search_memories" => {
            let args: SearchArgs = match serde_json::from_value(call.arguments.clone()) {
                Ok(a) => a,
                Err(e) => return err_result(call, format!("invalid args: {e}")),
            };
            // tier "all" treated as None (no filter)
            let tier_filter = match args.tier.as_deref() {
                Some("all") | None => None,
                Some(other) => Some(other),
            };
            // Use recall helper if available; else fallback to store.search_similar.
            let results = match crate::intelligence::recall::recall(
                ctx.store.clone(),
                &args.query,
                args.limit,
                tier_filter,
            ).await {
                Ok(r) => r,
                Err(e) => return err_result(call, format!("search failed: {e}")),
            };
            let payload: Vec<Value> = results.iter().map(|m| {
                let snippet: String = m.content.chars().take(200).collect();
                json!({
                    "id": m.id,
                    "content_snippet": snippet,
                    "tier": m.knowledge_tier.clone().unwrap_or_else(|| "explicit".into()),
                })
            }).collect();
            // Track retrieved IDs as "read_but_discarded" by default; select_final_memories promotes them.
            if let Ok(mut rbd) = ctx.read_but_discarded.lock() {
                for m in &results { rbd.insert(m.id.clone()); }
            }
            ok_result(call, serde_json::to_string(&payload).unwrap_or("[]".into()))
        }
        "create_memory" => {
            let args: CreateArgs = match serde_json::from_value(call.arguments.clone()) {
                Ok(a) => a,
                Err(e) => return err_result(call, format!("invalid args: {e}")),
            };
            let input = CreateMemory {
                content: args.content,
                knowledge_tier: Some(args.knowledge_tier),
                source_ids: args.source_ids.clone(),
                tags: args.tags,
                ..Default::default()
            };
            match ctx.store.store_with_outcome(input).await {
                Ok(outcome) => {
                    let id = match &outcome {
                        crate::storage::store::StoreOutcome::Stored(m) => m.id.clone(),
                        crate::storage::store::StoreOutcome::Deduplicated { existing_id, .. } => existing_id.clone(),
                        _ => "unknown".into(),
                    };
                    // REAS-10: create_memory source_ids get ×1.3 boost at loop terminal.
                    if let Some(src_ids) = args.source_ids.as_ref() {
                        if let Ok(mut fs) = ctx.final_selection.lock() {
                            for s in src_ids { fs.insert(s.clone()); }
                        }
                    }
                    ok_result(call, json!({"id": id}).to_string())
                }
                Err(e) => err_result(call, format!("create_memory: {e}")),
            }
        }
        "update_memory" => {
            let args: UpdateArgs = match serde_json::from_value(call.arguments.clone()) {
                Ok(a) => a,
                Err(e) => return err_result(call, format!("invalid args: {e}")),
            };
            let update = UpdateMemory {
                content: args.content,
                tags: args.tags,
                ..Default::default()
            };
            match ctx.store.update(&args.id, update).await {
                Ok(_) => ok_result(call, json!({"updated": args.id}).to_string()),
                Err(e) => err_result(call, format!("update_memory: {e}")),
            }
        }
        "delete_memory" => {
            let args: DeleteArgs = match serde_json::from_value(call.arguments.clone()) {
                Ok(a) => a,
                Err(e) => return err_result(call, format!("invalid args: {e}")),
            };
            // Phase 24 D-06 guard.
            match ctx.store.is_source_of_any_derived(&args.id).await {
                Ok(true) => return err_result(call, format!(
                    "cannot delete: memory {} is a source of one or more derived memories (Phase 24 D-06 — cascade-delete forbidden)",
                    args.id
                )),
                Ok(false) => {}
                Err(e) => return err_result(call, format!("delete guard check: {e}")),
            }
            match ctx.store.delete(&args.id).await {
                Ok(()) => ok_result(call, json!({"deleted": args.id}).to_string()),
                Err(e) => err_result(call, format!("delete_memory: {e}")),
            }
        }
        "annotate_memory" => {
            let args: AnnotateArgs = match serde_json::from_value(call.arguments.clone()) {
                Ok(a) => a,
                Err(e) => return err_result(call, format!("invalid args: {e}")),
            };
            match ctx.store.add_annotation(&args.id, &args.annotation).await {
                Ok(()) => ok_result(call, json!({"annotated": args.id}).to_string()),
                Err(e) => err_result(call, format!("annotate_memory: {e}")),
            }
        }
        "select_final_memories" => {
            let args: SelectArgs = match serde_json::from_value(call.arguments.clone()) {
                Ok(a) => a,
                Err(e) => return err_result(call, format!("invalid args: {e}")),
            };
            if let Ok(mut fs) = ctx.final_selection.lock() {
                for id in &args.ids { fs.insert(id.clone()); }
            }
            // Remove from read_but_discarded — they're now selected.
            if let Ok(mut rbd) = ctx.read_but_discarded.lock() {
                for id in &args.ids { rbd.remove(id); }
            }
            ok_result(call, json!({"selected": args.ids.len()}).to_string())
        }
        other => err_result(call, format!("unknown tool: {other}")),
    }
}
